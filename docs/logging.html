    timestamp = Date.now()

      newMessage = { id, sender, message, destination, timestamp }

Log and Forward

Log a message and send it out via a Now.js function. I found myself writing these three lines often and got tired of it.

 
  logAndForward: (sender, message, destination={'room':'itp'}, callback) ->
    timestamp = Date.now()
    @logMessage sender, message, destination
    callback(timestamp, sender, message)

module.exports = logging
